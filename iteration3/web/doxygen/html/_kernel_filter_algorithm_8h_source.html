110  static double** kernelSharpen(int &size, float *args) {
111  KernelFilter kernel;
112  int t_size = 60 / args[0];
113  if (t_size > 10) t_size = 5;
114  kernel.setSize(t_size);
115  size = kernel.getSize();
116  // We will be manually ensuring the brightness does not change
117  // so pull the double version right away
118  double **filter = kernel.factoredFilter();
119  // Balance num b = perimeter of the matrix +1
120  // so the total is is 1 when it is set
121  int b = size * 4 - 3;
122  // Remainder r = the amount of b remaining after movement from center
123  double r = b;
124  // f_val the amount to fill each extra square to
125  double f_val = 7 / args[0];
126  // Assign values
127  for (int i = 0; i < size; ++i) {
128  for (int j = 0; j < size; ++j)
129  {
130  if (i == 0 || j == 0 || i == (size - 1) || j == (size - 1))
131  filter[i][j] = -1.0; // negative edges
132  else if (i == j && i == size/2)
133  continue; // skip center for later assignment
134  else {
135  filter[i][j] = f_val;
136  r -= f_val;
137  }
138  }
139  }
140  filter[size/2][size/2] = r;
141  return filter;
